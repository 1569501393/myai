## Python 使用示例

```python
import requests
from requests.auth import HTTPBasicAuth

SAP_HOST = "mysap.goodsap.cn:50400"
AUTH = HTTPBasicAuth("U1170", "hysoft888999")

def get_ztjq0001():
    url = f"http://{SAP_HOST}/sap/bc/zzjq"
    response = requests.get(
        url,
        auth=AUTH,
        params={"table": "ZTJQ0001"},
        headers={"sap-client": "400"}
    )
    return response.json()

# 获取所有数据
data = get_ztjq0001()
for row in data:
    print(f"SID: {row['SID']}, Name: {row['NAME']}, Sex: {row['SEX']}")
```
