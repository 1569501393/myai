import { createApp } from 'https://unpkg.com/vue@3/dist/vue.esm-browser.js' const API_BASE = 'http://localhost:8888' const app = createApp({ data() { return { students: [], loading: false, showModal: false, modalMode: 'add', search: '', pagination: { page: 1, perPage: 10, total: 0 }, form: { id: null, student_no: '', name: '', gender: '男', birth_date: '', phone: '', email: '', address: '' }, detail: null, showDetail: false } }, computed: { totalPages() { return Math.ceil(this.pagination.total / this.pagination.perPage) } }, methods: { async fetchStudents() { this.loading = true try { const params = new URLSearchParams({ page: this.pagination.page, per_page: this.pagination.perPage, search: this.search }) const res = await fetch(`${API_BASE}/students?${params}`) const json = await res.json() if (json.success) { this.students = json.data.data this.pagination.total = json.data.total } } catch (e) { alert('加载失败: ' + e.message) } finally { this.loading = false } }, openAdd() { this.modalMode = 'add' this.form = { id: null, student_no: '', name: '', gender: '男', birth_date: '', phone: '', email: '', address: '' } this.showModal = true }, openEdit(student) { this.modalMode = 'edit' this.form = { ...student } this.showModal = true }, async save() { try { const url = this.modalMode === 'add' ? `${API_BASE}/students` : `${API_BASE}/students/${this.form.id}` const method = this.modalMode === 'add' ? 'POST' : 'PUT' const res = await fetch(url, { method, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(this.form) }) const json = await res.json() if (json.success) { this.showModal = false this.fetchStudents() } else { alert(json.message) } } catch (e) { alert('保存失败: ' + e.message) } }, async deleteStudent(id) { if (!confirm('确定要删除该学生吗？')) return try { const res = await fetch(`${API_BASE}/students/${id}`, { method: 'DELETE' }) const json = await res.json() if (json.success) { this.fetchStudents() } else { alert(json.message) } } catch (e) { alert('删除失败: ' + e.message) } }, viewDetail(student) { this.detail = student this.showDetail = true }, closeDetail() { this.showDetail = false }, searchStudents() { this.pagination.page = 1 this.fetchStudents() }, goPage(page) { if (page >= 1 && page <= this.totalPages) { this.pagination.page = page this.fetchStudents() } } }, mounted() { this.fetchStudents() }, template: `

学生管理系统 MVP

加载中...
ID 学号 姓名 性别 出生日期 电话 操作
{{ s.id }} {{ s.student_no }} {{ s.name }} {{ s.gender }} {{ s.birth_date || '-' }} {{ s.phone || '-' }}
暂无数据
{{ pagination.page }} / {{ totalPages }}

{{ modalMode === 'add' ? '添加学生' : '编辑学生' }}

学生详情

学号: {{ detail.student_no }}
姓名: {{ detail.name }}
性别: {{ detail.gender }}
出生日期: {{ detail.birth_date || '-' }}
电话: {{ detail.phone || '-' }}
邮箱: {{ detail.email || '-' }}
地址: {{ detail.address || '-' }}
` }) app.mount('#app')
